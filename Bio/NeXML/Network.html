def to_xml
  node = @@writer.create_node( "network", @@writer.attributes( self, :id, :'xsi:type', :label ) )

  self.each_node do |n|
    node << n.to_xml
  end

  self.each_edge do |edge|
    node << edge.to_xml
  end

  node
end
