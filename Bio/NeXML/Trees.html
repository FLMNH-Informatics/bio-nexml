def to_xml
  node = @@writer.create_node( "trees", @@writer.attributes( self, :id, :label, :otus ) )

  self.each_tree do |tree|
    node << tree.to_xml
  end

  self.each_network do |network|
    node << network.to_xml
  end

  node
end
